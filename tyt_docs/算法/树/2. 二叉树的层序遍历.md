### 层序遍历：
#### 队列法
```python
  class Solution:
    def layerTraverse(root):
      if root: queue = collections.deque([root])
      result = []
      while queue:
        level = []
        for qu in range(len(queue)):
          cur = queue.popleft()
          level.append(cur.val)
          if cur.left:
            queue.append(cur.left)
          if cur.right:
            queue.append(cur.right)
        result.append(level)
      return result
```
#### 递归法
```python
  class Solution:
    def layerTraverse(root):

      if not root: return []

      levels = []
      def bfs(node, level):
        if not node: return 
        if len(levels) == level:
          levels.append([])
        levels[level].append(node.val)
        if node.left:
          bfs(node.left, level + 1)
        if node.right:
          bfs(node.right, level + 1)
      
      bfs(root, 0)
      return levels
```

**107.二叉树的层次遍历 II**
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def levelOrderBottom(self, root):
        """
        :type root: Optional[TreeNode]
        :rtype: List[List[int]]
        """
        if not root: return []
        levels = []
        def bfs(node, level):
            if not node: return
            if len(levels) == level: levels.append([])

            levels[level].append(node.val)

            bfs(node.left, level + 1)
            bfs(node.right, level + 1)

        bfs(root, 0)
        return levels[::-1]
```

**199.二叉树的右视图**
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
#  层序遍历，每一层的最右节点
class Solution(object):
    def rightSideView(self, root):
        """
        :type root: Optional[TreeNode]
        :rtype: List[int]
        """

        # 这里deque的时候需要判断root
        # if not root :return []
        # queue = deque([root])
        # 如上述来进行书写也可以哦！
        queue = deque([root]) if root else deque()
        result = []
        while queue:
            level = []

            for _ in range(len(queue)):
                node = queue.popleft()
                level.append(node.val)
                if node.left: queue.append(node.left)
                if node.right: queue.append(node.right)
            result.append(level[len(level)-1])
        return result
```

**637.二叉树的层平均值**
```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def averageOfLevels(self, root):
        """
        :type root: Optional[TreeNode]
        :rtype: List[float]
        """

        queue = deque([root]) if root else []
        result = []
        while queue:
            level = []
            for _ in range(len(queue)):
                node = queue.popleft()
                level.append(node.val)
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
            # Q: 这里为什么要用float(len(level))?
            # A: Python 2 中的整数除法规则（地板除 / 向下取整）,Python 3中是正常获得14.5的，这里是leetcode的老环境，可能使用的是python2.
            # Q：保留小数的方法有哪些？
            # A：1.round(数值，保留几位小数)。2.f"{数值:.5f}"。3.format(..., ".5f")。4.Decimal.quantize(...)。
            #  ✅ float（浮点型）
            result.append(round(sum(level) / float(len(level)), 5))
            # result.append(f"{(sum(level) / float(len(level))):.5f}") ❌str（字符串）
            # result.append(format(round(sum(level) / float(len(level))), ".5f")) ❌str（字符串）
            # result.append(decimal(sum(level) / float(len(level))).quantize(decimal("0.00000"))) ❌Decimal（高精度类型）
        return result
```

**429.N叉树的层序遍历**
```python
"""
# Definition for a Node.
class Node(object):
    def __init__(self, val=None, children=None):
        self.val = val
        self.children = children
"""

class Solution(object):
    def levelOrder(self, root):
        """
        :type root: Node
        :rtype: List[List[int]]
        """
        queue = deque([root]) if root else []
        result = []
        while queue:
            level = []
            for _ in range(len(queue)):
                node = queue.popleft()
                level.append(node.val)

                if node.children:
                    for child in node.children:
                        queue.append(child)
            result.append(level)
        return result
```
